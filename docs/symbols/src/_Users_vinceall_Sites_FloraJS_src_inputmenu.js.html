1 /*global exports, document */
  2 /**
  3  * Creates a new InputMenu object.
  4  * An Input Menu lists key strokes and other input available
  5  * for the user to interact with the system.
  6  *
  7  * @constructor
  8  *
  9  * @param {Object} [opt_options] Options.
 10  * @param {string} [opt_options.position = 'top left'] A text representation
 11  *    of the menu's location. Possible values are 'top left', 'top center', 'top right',
 12  *    'bottom left', 'bottom center', 'bottom right', 'center'.
 13  * @param {number} [opt_options.opacity = 0.75] The menu's opacity.
 14  * @param {number} [opt_options.color = [255, 255, 255]] The menu's color.
 15  * @param {string} [opt_options.borderWidth = '1px'] The menu's border width.
 16  * @param {string} [opt_options.borderStyle = 'solid'] The menu's border style.
 17  * @param {Array|string} [opt_options.borderColor = 0.75] The menu's border color.
 18  */
 19 function InputMenu(opt_options) {
 20 
 21   'use strict';
 22 
 23   var me = this, options = opt_options || {};
 24 
 25   // if a world is not passed, use the first world in the universe
 26   this.world = options.world || exports.universe.first();
 27   this.position = options.position || 'top left';
 28   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 29   this.color = options.color || [255, 255, 255];
 30   this.borderWidth = options.borderWidth || '1px';
 31   this.borderStyle = options.borderStyle || 'solid';
 32   this.borderColor = options.borderColor || [204, 204, 204];
 33   this.colorMode = options.colorMode || 'rgb';
 34 
 35   if (exports.System.supportedFeatures.touch) {
 36     this.text =  exports.config.touchMap.stats + '-finger tap = stats | ' +
 37         exports.config.touchMap.pause + '-finger tap = pause | ' +
 38         exports.config.touchMap.reset + '-finger tap = reset';
 39   } else {
 40     this.text = '\'' + String.fromCharCode(exports.config.keyMap.pause).toLowerCase() + '\' = pause | ' +
 41       '\'' + String.fromCharCode(exports.config.keyMap.reset).toLowerCase() + '\' = reset | ' +
 42       '\'' + String.fromCharCode(exports.config.keyMap.stats).toLowerCase() + '\' = stats';
 43   }
 44 
 45   /**
 46    * Holds a reference to the caption's DOM elements.
 47    * @private
 48    */
 49   this._el = document.createElement('div');
 50   this._el.id = 'inputMenu';
 51   this._el.className = 'inputMenu ' + this.position;
 52   this._el.style.opacity = this.opacity;
 53   this._el.style.color = this.colorMode + '(' + this.color[0] + ', ' + this.color[1] +
 54         ', ' + this.color[2] + ')';
 55   this._el.style.borderWidth = this.borderWidth;
 56   this._el.style.borderStyle = this.borderStyle;
 57   if (typeof this.borderColor === 'string') {
 58     this._el.style.borderColor = this.borderColor;
 59   } else {
 60     this._el.style.borderColor = this.colorMode + '(' + this.borderColor[0] + ', ' + this.borderColor[1] +
 61         ', ' + this.borderColor[2] + ')';
 62   }
 63   this._el.appendChild(document.createTextNode(this.text));
 64   if (document.getElementById('inputMenu')) {
 65     document.getElementById('inputMenu').parentNode.removeChild(document.getElementById('inputMenu'));
 66   }
 67 
 68   if (exports.System.supportedFeatures.touch) {
 69     exports.Utils.addEvent(this._el, 'touchstart', function() {
 70       me.destroy();
 71     });
 72   } else {
 73     exports.Utils.addEvent(this._el, 'mouseup', function() {
 74       me.destroy();
 75     });
 76   }
 77 
 78   this.world.el.appendChild(this._el);
 79 }
 80 
 81 InputMenu.prototype.name = 'InputMenu';
 82 
 83 /**
 84  * Removes the menu's DOM element.
 85  */
 86 InputMenu.prototype.destroy = function() {
 87 
 88   'use strict';
 89 
 90   this._el.parentNode.removeChild(this._el);
 91 };
 92 
 93 exports.InputMenu = InputMenu;
