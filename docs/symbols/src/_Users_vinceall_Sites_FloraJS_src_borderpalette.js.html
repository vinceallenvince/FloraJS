1 /*global exports */
  2 /**
  3  * Creates a new BorderPalette object.
  4  *
  5  * Use this class to create a palette of border styles.
  6  *
  7  * @constructor
  8  */
  9 function BorderPalette() {
 10 
 11   'use strict';
 12 
 13   /**
 14    * Holds a list of border styles.
 15    * @private
 16    */
 17   this._borders = [];
 18 }
 19 
 20 /**
 21  * Define a name property.
 22  */
 23 BorderPalette.prototype.name = 'borderpalette';
 24 
 25 /**
 26  * Adds a random number of the passed border style to the 'borders' array.
 27  *
 28  * @param {Object} options A set of required options
 29  *    that includes:
 30  *    options.min {number} The minimum number of styles to add.
 31  *    options.max {number} The maximum number of styles to add.
 32  *    options.style {string} The border style.
 33  */
 34 BorderPalette.prototype.addBorder = function(options) {
 35 
 36   'use strict';
 37 
 38   var requiredOptions = {
 39     min: 'number',
 40     max: 'number',
 41     style: 'string'
 42   }, i, ln;
 43 
 44   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 45 
 46     ln = exports.Utils.getRandomNumber(options.min, options.max);
 47 
 48     for (i = 0; i < ln; i++) {
 49       this._borders.push(options.style);
 50     }
 51   }
 52   return this;
 53 };
 54 
 55 /**
 56  * @returns A style randomly selected from the 'borders' property.
 57  * @throws {Error} If the 'borders' property is empty.
 58  */
 59 BorderPalette.prototype.getBorder = function() {
 60 
 61   'use strict';
 62 
 63   if (this._borders.length > 0) {
 64     return this._borders[exports.Utils.getRandomNumber(0, this._borders.length - 1)];
 65   } else {
 66     throw new Error('BorderPalette.getBorder: You must add borders via addBorder() before using getBorder().');
 67   }
 68 };
 69 exports.BorderPalette = BorderPalette;
