  2 /**
  3  * Creates a new World.
  4  *
  5  * @constructor
  6  *
  7  * @param {Object} [opt_options] World options.
  8  * @param {string} [opt_options.id = "m-" + World._idCount] An id. If an id is not provided, one is created.
  9  * @param {boolean} [opt_options.isStatic = true] Set to false if transforming the world every frame.
 10  * @param {number} [opt_options.clock = 0] Increments each frame.
 11  * @param {number} [opt_options.c = 0.01] Coefficient of friction.
 12  * @param {Object} [opt_options.gravity = {x: 0, y: 1}] Gravity
 13  * @param {Object} [opt_options.wind = {x: 0, y: 0}] Wind
 14  * @param {Object} [opt_options.location = {x: 0, y: 0}] Initial location
 15  * @param {number} [opt_options.scale = 1] Scale
 16  * @param {number} [opt_options.angle = 0] Angle
 17  * @param {number} [opt_options.opacity = 0.85] Opacity
 18  * @param {string} [opt_options.colorMode = 'rgb'] Color mode. Valid options are 'rgb'. 'hex' and 'hsl' coming soon.
 19  * @param {Array} [opt_options.color = null] The object's color expressed as an rbg or hsl value. ex: [255, 100, 0]
 20  * @param {number} [opt_options.borderWidth = 0] The border width.
 21  * @param {number} [opt_options.borderStyle = 0] The border style. Possible values: 'none', 'solid', 'dotted', 'dashed', 'double', 'inset', 'outset', 'groove', 'ridge'
 22  * @param {number} [opt_options.borderColor = null] The border color.
 23  * @param {number} [opt_options.borderRadius = 0] The border radius.
 24  * @param {number|string} [opt_options.boxShadow = 0] The box shadow.
 25  * @param {number} [opt_options.width = window.width] The world width.
 26  * @param {number} [opt_options.height = window.height] The world height.
 27  * @param {number} [opt_options.zIndex = 0] The world z-index.
 28  * @param {boolean} [opt_options.isTopDown = true] Set to true to orient the gravity vector when listening to the devicemotion event.
 29  * @param {number} [opt_options.compassHeading = 0] The compass heading. Value is set via the deviceorientation event.
 30  * @param {number} [opt_options.compassAccuracy = 0] The compass accuracy. Value is set via the deviceorientation event.
 31  * @param {boolean} [opt_options.isDeviceMotion = false] Set to true add the devicemotion event listener. Typically use with accelerometer equipped devices.
 32  * @param {function} [opt_options.beforeStep = ''] A function to run before the step() function.
 33  * @param {function} [opt_options.afterStep = ''] A function to run after the step() function.
 34  */
 35 function World(opt_options) {
 36 
 37   'use strict';
 38 
 39   var me = this, options = opt_options || {},
 40       winSize = exports.Utils.getWindowSize();
 41 
 42   this.isStatic = options.isStatic || true;
 43   this.clock = options.clock || 0;
 44   this.c = options.c || 0.01;
 45   this.gravity = options.gravity || new exports.Vector(0, 1);
 46   this.wind =  options.wind || new exports.Vector();
 47   this.location = options.location || new exports.Vector();
 48 
 49   this.scale = options.scale || 1;
 50   this.angle = options.angle || 0;
 51   this.opacity = options.opacity || 1;
 52   this.colorMode = options.colorMode || 'rgb';
 53   this.color = options.color || [0, 0, 0];
 54   this.borderWidth = options.borderWidth || 0;
 55   this.borderStyle = options.borderStyle || 'none';
 56   this.borderColor = options.borderColor || null;
 57   this.borderRadius = options.borderRadius || 0;
 58   this.boxShadow = options.boxShadow || 0;
 59 
 60   this.zIndex = 0;
 61   this.isTopDown = true;
 62   this.compassHeading = 0;
 63   this.compassAccuracy = 0;
 64   this.isDeviceMotion = !options.isDeviceMotion;
 65 
 66   this.beforeStep = options.beforeStep || undefined;
 67   this.afterStep = options.afterStep || undefined;
 68 
 69   /**
 70    * If no DOM element is passed for the world,
 71    * use document.body. Because the body initially has
 72    * no height, we use the window height.
 73    */
 74   if (!options.el) {
 75     this.el = document.body; // if no world element is passed, use document.body
 76     this.width = winSize.width;
 77     this.height = winSize.height;
 78     this.id = World.name + "-" + World._idCount; // if no id, create one
 79   } else {
 80     this.el = options.el;
 81     this.id = this.el.id; // use the element's id as the record id
 82     this.width = this.el.offsetWidth;
 83     this.height = this.el.offsetHeight;
 84   }
 85 
 86   this.el.className = 'world floraElement';
 87   this.el.style.width = this.width + 'px';
 88   this.el.style.height = this.height + 'px';
 89 
 90   World._idCount += 1; // increment id
 91 
 92   // events
 93 
 94   exports.Utils.addEvent(window, 'resize', function(e) { // listens for window resize
 95     me.resize.call(me);
 96   });
 97 
 98   /*if (window.addEventListener && this.isDeviceMotion) {
 99     window.addEventListener("devicemotion", function(e) { // listens for device motion events
100       me.devicemotion.call(me, e);
101     }, false);
102   }
103 
104   if (window.addEventListener && this.isDeviceOrientation) {
105     window.addEventListener("deviceorientation", function(e) { // listens for device orientation events
106       me.deviceorientation.call(me, e);
107     }, false);
108   }*/
109 }

123 /**

128 World.prototype.configure = function(opt_el) { // should be called after doc ready()
129 
130   'use strict';
131 
132   var el = opt_el || null;
133 
134   this.el = el || document.body; // if no world element is passed, use document.body
135   this.el.style.width = this.width + 'px';
136   this.el.style.height = this.height + 'px';

139 /**
140  * Updates a world instance with passed arguments.
141  * Typically called to change the window's default size or
142  * change the world's style.
143  *
144  * @param {Object} props A hash of properties to update.
145  *    Optional properties:
146  *    opt_props.style A map of css styles to apply.
147  */
148 World.prototype.update = function(opt_props) {
149 
150   'use strict';
151 
152   var i, key, cssText = '',
153       props = exports.Interface.getDataType(opt_props) === 'object' ? opt_props : {};
154 
155   for (key in props) {
156     if (props.hasOwnProperty(key)) {
157       this[key] = props[key];
158     }
159   }
160 
161   if (props.el) { // if updating the world's DOM element; update the width and height
162     this.width = parseInt(this.el.style.width.replace('px', ''), 10);
163     this.height = parseInt(this.el.style.height.replace('px', ''), 10);
164   }
165 
166   if (!this.el.style.setAttribute) { // WC3
167     if (props.style) {
168       for (i in props.style) {
169         if (props.style.hasOwnProperty(i)) {
170           this.el.style[i] = props.style[i];
171           cssText = cssText + i + ': ' + props.style[i] + ';';
172         }
173       }
174     }
175   } else { // IE
176     if (props.style) {
177       for (i in props.style) {
178         if (props.style.hasOwnProperty(i)) {
179           cssText = cssText + i + ': ' + props.style[i] + ';';
180         }
181       }
182     }
183     this.el.style.setAttribute('cssText', cssText, 0);
184   }
185 };

187 /**
188  * Called from a window resize event, resize() repositions all Flora elements relative
189  * to the new window size. Also, if the world is the document.body, resets the body's
190  * width and height attributes.
191  */
192 World.prototype.resize = function() {
193 
194   'use strict';
195 
196   var i, max, elementLoc, controlCamera, winSize = exports.Utils.getWindowSize(),
197     windowWidth = winSize.width,
198     windowHeight = winSize.height,
199     elements = exports.elementList.all();
200 
201   // check of any elements control the camera
202   for (i = 0, max = elements.length; i < max; i += 1) {
203     if (elements[i].controlCamera) {
204       controlCamera = true;
205       break;
206     }
207   }
208 
209   // loop thru elements
210   if (!controlCamera) {
211     for (i = 0, max = elements.length; i < max; i += 1) {
212 
213       elementLoc = elements[i].location; // recalculate location
214 
215       elementLoc.x = windowWidth * (elementLoc.x/this.width);
216       elementLoc.y = windowHeight * (elementLoc.y/this.height);
217 
218     }
219 
220     if (this.el === document.body) {
221       this.width = windowWidth;
222       this.height = windowHeight;
223     }
224   }
225 };

227 /**
228  * Called from a window devicemotion event, updates the world's gravity
229  * relative to the accelerometer's values.
230  */
231 World.prototype.devicemotion = function(e) {
232 
233   'use strict';
234 
235   if (window.orientation === 0) {
236     if (this.isTopDown) {
237       this.gravity = new exports.Vector(e.accelerationIncludingGravity.x, e.accelerationIncludingGravity.y * -1); // portrait
238     } else {
239       this.gravity = new exports.Vector(e.accelerationIncludingGravity.x, (e.accelerationIncludingGravity.z + 7.5) * 2); // portrait 45 degree angle
240     }
241   } else if (window.orientation === -90) {
242     this.gravity = new exports.Vector(e.accelerationIncludingGravity.y, e.accelerationIncludingGravity.x );
243   } else {
244     this.gravity = new exports.Vector(e.accelerationIncludingGravity.y * -1, e.accelerationIncludingGravity.x * -1);
245   }
246 
247   if (this.showDeviceOrientation) {
248     document.getElementById('compassDisplay').val("orientation: " + window.orientation + " x: " +
249         e.accelerationIncludingGravity.x.toFixed(2) + " y: " + e.accelerationIncludingGravity.y.toFixed(2) + " z: " +
250         e.accelerationIncludingGravity.z.toFixed(2));
251   }
252 };

254 /**
255  * Called from a window deviceorientation event, updates the world's compass values.
256  *
257  * @param {Object} e An event object passed from the event listener.
258  */
259 World.prototype.deviceorientation = function(e) {
260 
261   'use strict';
262 
263   var compassHeading = e.webkitCompassHeading,
264     compassAccuracy = e.webkitCompassAccuracy;
265 
266   this.compassAccuracy = compassAccuracy;
267   this.compassHeading = compassHeading;
268 
269   if (this.showCompassHeading) {
270     document.getElementById('compassDisplay').val("heading: " + compassHeading.toFixed(2) +
271         " accuracy: +/- " +compassAccuracy);
272   }
273 };

276 /**
277  * Called every frame, step() updates the world's properties.
278  */
279 World.prototype.step = function() {
280 
281   'use strict';
282 
283   if (this.beforeStep) {
284     this.beforeStep.apply(this);
285   }
286   if (this.afterStep) {
287     this.afterStep.apply(this);
288   }
289 };

291 /**
292  * Called every frame, draw() renders the world.
293  */
294 World.prototype.draw = function() {
295 
296   'use strict';
297 
298   /**
299    * If there's not an object controlling the camera,
300    * we want to draw the world once.
301    */
302   if (!exports.camera.controlObj && this.isStatic && this.clock > 0) {
303     return;
304   }
305 
306   this.el.style.cssText = exports.Utils.getCSSText({
307     x: this.location.x,
308     y: this.location.y,
309     s: this.scale,
310     a: this.angle,
311     o: this.opacity,
312     w: this.width,
313     h: this.height,
314     cm: this.colorMode,
315     color: this.color,
316     z: this.zIndex,
317     borderWidth: this.borderWidth,
318     borderStyle: this.borderStyle,
319     borderColor: this.borderColor,
320     borderRadius: this.borderRadius,
321     boxShadow: this.boxShadow
322   });
323 };
