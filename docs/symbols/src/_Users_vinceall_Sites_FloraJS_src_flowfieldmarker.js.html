  2 /**
  3  * Creates a new FlowFieldMarker.
  4  *
  5  * @constructor
  6  * @param {Object} options Options.
  7  * @param {Object} options.location Location.
  8  * @param {number} options.scale Scale.
  9  * @param {number} options.opacity Opacity
 10  * @param {number} options.width Width.
 11  * @param {number} options.height Height.
 12  * @param {number} options.angle Angle.
 13  * @param {string} options.colorMode Color mode.
 14  * @param {Object} options.color Color.
 15  */
 16 function FlowFieldMarker(options) {
 17 
 18   'use strict';
 19 
 20   var requiredOptions = {
 21         location: 'object',
 22         scale: 'number',
 23         angle: 'number',
 24         opacity: 'number',
 25         width: 'number',
 26         height: 'number',
 27         colorMode: 'string',
 28         color: 'array'
 29       }, el, nose;
 30 
 31   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 32 
 33     el = document.createElement("div");
 34     nose = document.createElement("div");
 35     el.className = "flowFieldMarker floraElement";
 36     nose.className = "nose";
 37     el.appendChild(nose);
 38 
 39     el.style.cssText = exports.Utils.getCSSText({
 40       x: options.location.x - options.width/2,
 41       y: options.location.y - options.height/2,
 42       s: options.scale,
 43       a: options.angle,
 44       o: options.opacity,
 45       w: options.width,
 46       h: options.height,
 47       cm: options.colorMode,
 48       color: options.color,
 49       z: options.zIndex,
 50       borderWidth: options.borderWidth,
 51       borderStyle: options.borderStyle,
 52       borderColor: options.borderColor,
 53       borderRadius: options.borderRadius,
 54       boxShadow: options.boxShadow
 55     });
 56 
 57     return el;
 58   }
 59 }
 60 
 61 /**
 62  * Define a name property.
 63  */
 64 FlowFieldMarker.prototype.name = 'flowfieldmarker';
 65 
 66 exports.FlowFieldMarker = FlowFieldMarker;
