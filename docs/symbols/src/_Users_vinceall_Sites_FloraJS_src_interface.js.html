1 /*global console */
  2 /** @namespace */
  3 var Interface = {};
  4 
  5 /**
  6  * @description Compares passed options to a set of required options.
  7  * @param {Object} optionsPassed An object containing options passed to a function.
  8  * @param {Object} optionsRequired An object containing a function's required options.
  9  * @param {string=} opt_from A message or identifier from the calling function.
 10  * @returns {boolean} Returns true if all required options are present and of the right data type.
 11  *     Returns false if any required options are missing or are the wrong data type. Also returns false
 12  *     if any of the passed options are empty.
 13  * @example
 14  * optionsPassed = {
 15  *  form: document.getElementById("form"),
 16  *    color: "blue",
 17  *  total: "a whole bunch",
 18  *   names: ["jimmy", "joey"]
 19  * }
 20  * optionsRequired = {
 21  *   form: "object",
 22  *   color: "string",
 23  *   total: "number|string",
 24  *   names: "array"
 25  * }
 26  * checkRequiredOptions(optionsPassed, optionsRequired) returns true in this case.
 27  *
 28  * Notice you can separate required options data types with a pipe character to allow multiple data types.
 29  */
 30 Interface.checkRequiredParams = function(optionsPassed, optionsRequired, opt_from) {
 31 
 32   'use strict';
 33 
 34   var i, msg, check = true;
 35 
 36   for (i in optionsRequired) { // loop thru required options
 37 
 38     if (optionsRequired.hasOwnProperty(i)) {
 39 
 40       try {
 41 
 42         // if no options were passed
 43         if (typeof optionsPassed === "undefined") {
 44           throw new Error('checkRequiredOptions: No options were passed.');
 45         }
 46 
 47         // if there is not a corresponding key in the passed options; or options passed value is blank
 48         if (!this.checkDataType(this.getDataType(optionsPassed[i]), optionsRequired[i].split('|')) || optionsPassed[i] === "") {
 49 
 50           check = false;
 51 
 52           if (optionsPassed[i] === '') {
 53             msg = 'checkRequiredOptions: required option "' + i + '" is empty.';
 54           } else if (typeof optionsPassed[i] === 'undefined') {
 55             msg = 'checkRequiredOptions: required option "' + i + '" is missing from passed options.';
 56           } else {
 57             msg = 'checkRequiredOptions: passed option "' + i + '" must be type ' + optionsRequired[i] +
 58             '. Passed as ' + this.getDataType(optionsPassed[i]) + '.';
 59           }
 60 
 61           throw new Error(msg);
 62         }
 63       } catch (err) {
 64         if (typeof console !== 'undefined') {
 65           console.log('ERROR: ' + err.message + (opt_from ? ' from: ' + opt_from : ''));
 66         }
 67       }
 68     }
 69   }
 70   return check;
 71 };
 72 
 73 /**
 74  * Loops through an array of data types and checks if the
 75  * passed option matches any entry.
 76  *
 77  * @param {string} option The data type to check.
 78  * @param {array} typesToMatch An array of data types to check option against.
 79  * @returns {boolean} If option matches any entry in typesToMatch,
 80  * return true. Else, returns false.
 81  */
 82 Interface.checkDataType = function(option, typesToMatch) {
 83 
 84   'use strict';
 85 
 86   var i, max;
 87 
 88   for (i = 0, max = typesToMatch.length; i < max; i++) {
 89     if (option === typesToMatch[i]) {
 90       return true;
 91     }
 92   }
 93   return false;
 94 };
 95 
 96 /**
 97  * Checks for data type.
 98  *
 99  * @returns {string} The data type of the passed variable.
100  */
101 Interface.getDataType = function(element) {
102 
103   'use strict';
104 
105   if (Object.prototype.toString.call(element) === '[object Array]') {
106     return 'array';
107   }
108 
109   if (Object.prototype.toString.call(element) === '[object NodeList]') {
110     return 'nodeList';
111   }
112 
113   return typeof element;
114 };
