  2 /**
  3  * Creates a new Element. All Flora elements extend Element. Element holds the minimum properties
  4  * need to render the element via draw().
  5  *
  6  * @constructor
  7  *
  8  * @param {Object} [opt_options] Element options.
  9  * @param {string} [opt_options.id = "m-" + Element._idCount] An id. If an id is not provided, one is created.
 10  * @param {Object|function} [opt_options.view] HTML representing the Element instance.
 11  * @param {string} [opt_options.className = 'agent'] The corresponding DOM element's class name.
 12  * @param {array} [opt_options.sensors = []] A list of sensors attached to this object.
 13  * @param {boolean} [opt_options.controlCamera = false] If true, camera will follow this object.
 14  * @param {number} [opt_options.width = 20] Width
 15  * @param {number} [opt_options.height = 20] Height
 16  * @param {number} [opt_options.scale = 1] Scale
 17  * @param {number} [opt_options.angle = 0] Angle
 18  * @param {number} [opt_options.opacity = 0.85] Opacity
 19  * @param {string} [opt_options.colorMode = 'rgb'] Color mode. Valid options are 'rgb'. 'hex' and 'hsl' coming soon.
 20  * @param {Array} [opt_options.color = null] The object's color expressed as an rbg or hsl value. ex: [255, 100, 0]
 21  * @param {number} [opt_options.zIndex = 1] z-index
 22  * @param {number} [opt_options.borderWidth = false] borderWidth The object's border width in pixels.
 23  * @param {string} [opt_options.borderStyle = false] borderStyle The object's border style. Valid options are 'none',
 24  *    'hidden', 'dotted', 'dashed', 'solid', 'double', 'groove', 'ridge', 'inset', 'outset', 'inherit'.
 25  * @param {string} [opt_options.borderColor = false] borderColor The object's border color expressed as an
 26  *    rbg or hsl value. ex: [255, 100, 0]
 27  * @param {string} [opt_options.borderRadius = false] borderRadius The object's border radius as a percentage.
 28  * @param {string} [opt_options.boxShadow =  false] boxShadow The object's box shadow.
 29  * @param {Object} [opt_options.location = The center of the world] The object's initial location.
 30  * @param {Object} [opt_options.acceleration = {x: 0, y: 0}] The object's initial acceleration.
 31  * @param {Object} [opt_options.velocity = {x: 0, y: 0}] The object's initial velocity.
 32  */
 33 function Element(opt_options) {
 34 
 35   'use strict';
 36 
 37   var i, max, key, prop;
 38 
 39   for (i = 0, max = arguments.length; i < max; i += 1) {
 40     prop = arguments[i];
 41     for (key in prop) {
 42       if (prop.hasOwnProperty(key)) {
 43         this[key] = prop[key];
 44       }
 45     }
 46   }
 47 
 48   var options = opt_options || {},
 49       elements = exports.elementList.all() || [],
 50       liquids = exports.liquids || [],
 51       repellers = exports.repellers || [],
 52       attractors = exports.attractors || [],
 53       heats = exports.heats || [],
 54       colds = exports.colds || [],
 55       predators = exports.predators || [],
 56       lights = exports.lights || [],
 57       oxygen = exports.oxygen || [],
 58       food = exports.food || [],
 59       world, constructorName = this.name || 'anon';
 60 
 61   this.id = options.id || constructorName.toLowerCase() + "-" + Element._idCount; // if no id, create one
 62 
 63   if (options.view && exports.Interface.getDataType(options.view) === "function") { // if view is supplied and is a function
 64     this.el = options.view.apply(this, options.viewArgs);
 65   } else if (exports.Interface.getDataType(options.view) === "object") { // if view is supplied and is an object
 66     this.el = options.view;
 67   } else {
 68     this.el = document.createElement("div");
 69   }
 70 
 71   // if a world is not passed, use the first world in the universe
 72   this.world = options.world || exports.universe.first();
 73   world = this.world;
 74 
 75   // set render properties
 76   this.location = options.location || new exports.Vector(world.width/2, world.height/2);
 77   this.acceleration = options.acceleration || new exports.Vector();
 78   this.velocity = options.velocity || new exports.Vector();
 79   this.width = options.width === 0 ? 0 : options.width || 20;
 80   this.height = options.height === 0 ? 0 : options.height || 20;
 81   this.scale = options.scale === 0 ? 0 : options.scale || 1;
 82   this.angle = options.angle || 0;
 83   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.85;
 84   this.colorMode = options.colorMode || 'rgb';
 85   this.color = options.color || null;
 86   this.zIndex = options.zIndex === 0 ? 0 : options.zIndex || 1;
 87   this.borderWidth = options.borderWidth || null;
 88   this.borderStyle = options.borderStyle || null;
 89   this.borderColor = options.borderColor || null;
 90   this.borderRadius = options.borderRadius || null;
 91   this.boxShadow = options.boxShadow || null;
 92 
 93     // set sensors
 94   this.sensors = options.sensors || [];
 95 
 96   this.className = options.className || constructorName.toLowerCase();
 97   this.className += ' floraElement';
 98 
 99   elements.push(this); // push new instance of Element
100 
101   this.el.id = this.id;
102   this.el.className = this.sensors.length > 0 ? (this.className + ' hasSensor') : this.className;
103   this.el.style.display = 'none';
104 
105   if (world.el) {
106     world.el.appendChild(this.el); // append the view to the World
107   }
108 
109   Element._idCount += 1; // increment id
110 
111   if (this.className.search('liquid') !== -1) {
112     liquids.push(this); // push new instance of liquids to liquid list
113   } else if (this.className.search('repeller') !== -1) {
114     repellers.push(this); // push new instance of repeller to repeller list
115   } else if (this.className.search('attractor') !== -1) {
116     attractors.push(this); // push new instance of attractor to attractor list
117   } else if (this.className.search('heat') !== -1) {
118     heats.push(this);
119   } else if (this.className.search('cold') !== -1) {
120     colds.push(this);
121   } else if (this.className.search('predator') !== -1) {
122     predators.push(this);
123   } else if (this.className.search('light') !== -1) {
124     lights.push(this);
125   } else if (this.className.search('oxygen') !== -1) {
126     oxygen.push(this);
127   } else if (this.className.search('food') !== -1) {
128     food.push(this);
129   }
130 
131   // setup camera control
132   this.controlCamera = !!options.controlCamera;
133 
134   if (this.controlCamera) { // if this object controls the camera
135 
136     exports.camera.controlObj = this;
137 
138     // need to position world so controlObj is centered on screen
139     world.location.x = -world.width/2 + (exports.Utils.getWindowSize().width)/2 + (world.width/2 - this.location.x);
140     world.location.y = -world.height/2 + (exports.Utils.getWindowSize().height)/2 + (world.height/2 - this.location.y);
141   }
142 }

161 /**
162  * Called by a mouseover event listener.
163  *
164  * @param {Object} e The event object passed by the listener.
165  * @param {Object} obj The object associated with the event target.
166  */
167 Element.mouseover = function(e, obj) {
168 
169   'use strict';
170 
171   obj.isMouseOut = false;
172   clearInterval(obj.mouseOutInterval);
173 };
174 
175 /**
176  * Called by a mousedown event listener.
177  *
178  * @param {Object} e The event object passed by the listener.
179  * @param {Object} obj The object associated with the event target.
180  */
181 Element.mousedown = function(e, obj) {
182 
183   'use strict';
184 
185   var target = e.target;
186 
187   obj.isPressed = true;
188   obj.isMouseOut = false;
189   obj.offsetX = e.pageX - target.offsetLeft;
190   obj.offsetY = e.pageY - target.offsetTop;
191 };
192 
193 /**
194  * Called by a mousemove event listener.
195  *
196  * @param {Object} e The event object passed by the listener.
197  * @param {Object} obj The object associated with the event target.
198  */
199 Element.mousemove = function(e, obj) {
200 
201   'use strict';
202 
203   var x, y;
204 
205   if (obj.isPressed) {
206 
207     obj.isMouseOut = false;
208 
209     x = e.pageX - obj.world.el.offsetLeft;
210     y = e.pageY - obj.world.el.offsetTop;
211 
212     if (Element.mouseIsInsideWorld(obj.world)) {
213       obj.location = new exports.Vector(x, y);
214     } else {
215       obj.isPressed = false;
216     }
217   }
218 };
219 
220 /**
221  * Called by a mouseup event listener.
222  *
223  * @param {Object} e The event object passed by the listener.
224  * @param {Object} obj The object associated with the event target.
225  */
226 Element.mouseup = function(e, obj) {
227 
228   'use strict';
229 
230   obj.isPressed = false;
231 };
232 
233 /**
234  * Called by a mouseout event listener.
235  *
236  * @param {Object} e The event object passed by the listener.
237  * @param {Object} obj The object associated with the event target.
238  */
239 Element.mouseout = function(e, obj) {
240 
241   'use strict';
242 
243   var me = obj,
244     x, y;
245 
246   if (obj.isPressed) {
247 
248     obj.isMouseOut = true;
249 
250     obj.mouseOutInterval = setInterval(function () { // if mouse is too fast for block update, update via an interval until it catches up
251 
252       if (me.isPressed && me.isMouseOut) {
253 
254         x = exports.mouse.loc.x - me.world.el.offsetLeft;
255         y = exports.mouse.loc.y - me.world.el.offsetTop;
256 
257         me.location = new exports.Vector(x, y);
258       }
259     }, 16);
260   }
261 };
262 
263 /**
264  * Checks if mouse location is inside the world.
265  *
266  * @param {Object} world A Flora world.
267  * @returns {boolean} True if mouse is inside world; False if
268  *    mouse is outside world.
269  */
270 Element.mouseIsInsideWorld = function(world) {
271 
272   'use strict';
273 
274   var x = exports.mouse.loc.x,
275       y = exports.mouse.loc.y;
276 
277   if (world) {
278     if (x > world.location.x &&
279       x < world.location.x + world.width &&
280       y > world.location.y &&
281       y < world.location.y + world.height) {
282       return true;
283     }
284   }
285   return false;
286 };

294 /**
295  * Renders the element to the DOM. Called every frame.
296  */
297 Element.prototype.draw = function() {
298 
299   'use strict';
300 
301   var width = typeof this.width === 'number' ? this.width : this.el.offsetWidth, // !! no
302       height = typeof this.height === 'number' ? this.height : this.el.offsetHeight; // !! no
303 
304   this.el.style.cssText = exports.Utils.getCSSText({
305     x: this.location.x - width/2,
306     y: this.location.y - height/2,
307     s: this.scale,
308     a: this.angle,
309     o: this.opacity,
310     w: this.width,
311     h: this.height,
312     cm: this.colorMode,
313     color: this.color,
314     z: this.zIndex,
315     borderWidth: this.borderWidth,
316     borderStyle: this.borderStyle,
317     borderColor: this.borderColor,
318     borderRadius: this.borderRadius,
319     boxShadow: this.boxShadow
320   });
321 };
