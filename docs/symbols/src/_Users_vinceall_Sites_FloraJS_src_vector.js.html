  2 /**
  3  * Creates a new Vector.
  4  *
  5  * @param {number} [opt_x = 0] The x location.
  6  * @param {number} [opt_y = 0] The y location.
  7  * @constructor
  8  */
  9 function Vector(opt_x, opt_y) {
 10   'use strict';
 11   var x = opt_x || 0,
 12       y = opt_y || 0;
 13   this.x = x;
 14   this.y = y;
 15 }

 17 /**
 18  * Subtract two vectors. Uses clone to avoid affecting the values of the vectors.
 19  *
 20  * @returns {Object} A vector.
 21  */
 22 Vector.VectorSub = function(v1, v2) {
 23   'use strict';
 24   return new Vector(v1.x - v2.x, v1.y - v2.y);

 57 /**
 58  * Get the midpoint between two vectors. Uses clone to avoid affecting the values of the vectors.

 60  * @returns {Object} A vector.

 64   return this.VectorAdd(v1, v2).div(2); // midpoint = (v1 + v2)/2

 67 /**
 68  * Get the angle between two vectors.
 69  *
 70  * @returns {Object} A vector.
 71  */
 72 Vector.VectorAngleBetween = function(v1, v2) {
 73   'use strict';
 74   var dot = v1.dot(v2),
 75   theta = Math.acos(dot / (v1.mag() * v2.mag()));
 76   return theta;
 77 };

 79 /**
 80 * Returns an new vector with all properties and methods of the
 81 * old vector copied to the new vector's prototype.
 82 *
 83 * @returns {Object} A vector.
 84 */
 85 Vector.prototype.clone = function() {
 86   'use strict';
 87   function F() {}
 88   F.prototype = this;
 89   return new F;
 90 };

144 /**
145  * Calculates the magnitude of this vector.
146  *
147  * @returns {number} The vector's magnitude.
148  */
149 Vector.prototype.mag = function() {
150   'use strict';
151   return Math.sqrt((this.x * this.x) + (this.y * this.y));
152 };
153 
154 /**
155  * Limits the vector's magnitude.
156  *
157  * @param {number} high The upper bound of the vector's magnitude.
158  */
159 Vector.prototype.limit = function(high) {
160   'use strict';
161   if (this.mag() > high) {
162     this.normalize();
163     this.mult(high);
164   }
165   return this;
166 };
167 
168 Vector.prototype.limitLow = function(low) {
169   'use strict';
170   if (this.mag() < low) {
171     this.normalize();
172     this.mult(low);
173   }
174   return this;
175 };

177 /**
178  * Divides a vector by its magnitude to reduce its magnitude to 1.
179  * Typically used to retrieve the direction of the vector for later manipulation.
180  *
181  * @returns {Object} This vector.
182  */
183 Vector.prototype.normalize = function() {
184   'use strict';
185   var m = this.mag();
186   if (m !== 0) {
187     return this.div(m);
188   }
189 };

191 /**
192  * Calculates the distance between this vector and a passed vector.
193  *
194  * @param {Object} vector The target vector.
195  * @returns {Object} This vector.
196  */
197 Vector.prototype.distance = function(vector) {
198   'use strict';
199   return Math.sqrt(Math.pow(vector.x - this.x, 2) + Math.pow(vector.y - this.y, 2));
200 };

202 /**
203  * Rotates a vector using a passed angle in radians.
204  *
205  * @param {number} radians The angle to rotate in radians.
206  * @returns {Object} This vector.
207  */
208 Vector.prototype.rotate = function(radians) {
209   'use strict';
210   var cos = Math.cos(radians),
211     sin = Math.sin(radians),
212     x = this.x,
213     y = this.y;
214 
215   this.x = x * cos - y * sin;
216   this.y = x * sin + y * cos;
217   return this;
218 };

222  *
223  * @param {Object} v1 The first vector.
224  * @param {Object} v1 The second vector.
225  * @returns {Object} A vector representing the midpoint between the passed vectors.
226  */
227 Vector.prototype.midpoint = function(v1, v2) {
228   'use strict';
229   return this.VectorAdd(v1, v2).div(2);

232 /**
233  * Calulates the dot product.
234  *
235  * @param {Object} vector The target vector.
236  * @returns {Object} A vector.
237  */
238 Vector.prototype.dot = function(vector) {
239   'use strict';
240   if (this.z && vector.z) {
241     return this.x * vector.x + this.y * vector.y + this.z * vector.z;
242   }
243   return this.x * vector.x + this.y * vector.y;
244 };
