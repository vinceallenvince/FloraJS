  2 /**
  3  * Creates a new FlowField.
  4  *
  5  * @constructor
  6  * @param {Object} [opt_options] Options.
  7  * @param {number} [opt_options.resolution = 50] The lower the value, the more vectors are created to define the flow field. Low values increase processing time to create the field.
  8  * @param {number} [opt_options.perlinSpeed = 0.01] The speed to move through the Perlin Noise space.
  9  * @param {number} [opt_options.perlinTime = 100] Sets the Perlin Noise time.
 10  * @param {Object} [opt_options.field = null] A list of vectors that define the flow field.
 11  * @param {Object} [opt_options.createMarkers = false] Set to true to visualize the flow field.
 12  */
 13 function FlowField(opt_options) {
 14 
 15   'use strict';
 16 
 17   var options = opt_options || {};
 18 
 19   this.resolution = options.resolution || 50;
 20   this.perlinSpeed = options.perlinSpeed || 0.01;
 21   this.perlinTime = options.perlinTime || 100;
 22   this.field = options.field || null;
 23   this.createMarkers = options.createMarkers || false;
 24   // if a world is not passed, use the first world in the universe
 25   this.world = options.world || exports.universe.first();
 26 }
 27 
 28 FlowField.prototype.name = 'FlowField';
 29 
 30 /**
 31  * Builds a FlowField.
 32  */
 33 FlowField.prototype.build = function() {
 34 
 35   'use strict';
 36 
 37   var i, max, col, colMax, row, rowMax, x, y, theta, fieldX, fieldY, field, angle,
 38       vectorList = {},
 39       world = this.world,
 40       cols = Math.ceil(world.width/parseFloat(this.resolution)),
 41       rows = Math.ceil(world.height/parseFloat(this.resolution)),
 42       xoff = this.perlinTime, // create markers and vectors
 43       yoff;
 44 
 45   for (col = 0, colMax = cols; col < colMax ; col += 1) {
 46     yoff = this.perlinTime;
 47     vectorList[col] = {};
 48     for (row = 0, rowMax = rows; row < rowMax ; row += 1) {
 49 
 50       x = col * this.resolution + this.resolution / 2; // get location on the grid
 51       y = row * this.resolution + this.resolution / 2;
 52 
 53       theta = exports.Utils.map(exports.SimplexNoise.noise(xoff, yoff, 0.1), 0, 1, 0, Math.PI * 2); // get the vector based on Perlin noise
 54       fieldX = Math.cos(theta);
 55       fieldY = Math.sin(theta);
 56       field = new exports.Vector(fieldX, fieldY);
 57       vectorList[col][row] = field;
 58       angle = exports.Utils.radiansToDegrees(Math.atan2(fieldY, fieldX)); // get the angle of the vector
 59 
 60       if (this.createMarkers) {
 61 
 62         var ffm = new exports.FlowFieldMarker({ // create the marker
 63           location: new exports.Vector(x, y),
 64           scale: 1,
 65           opacity: exports.Utils.map(angle, -360, 360, 0.1, 1),
 66           width: this.resolution,
 67           height: this.resolution/2,
 68           field: field,
 69           angle: angle,
 70           colorMode: 'rgb',
 71           color: [200, 100, 50]
 72         });
 73         world.el.appendChild(ffm);
 74       }
 75       yoff += parseFloat(this.perlinSpeed);
 76     }
 77     xoff += parseFloat(this.perlinSpeed);
 78   }
 79   this.field = vectorList;
 80 };
 81 
 82 exports.FlowField = FlowField;
