1 /*global exports */
  2 /**
  3  * Creates a new ElementList.
  4  *
  5  * @constructor
  6  * @param {Object} [opt_options] Options.
  7  */
  8 function ElementList(opt_options) {
  9 
 10   'use strict';
 11 
 12   var options = opt_options || {};
 13 
 14   /**
 15    * Holds a list of elements.
 16    * @private
 17    */
 18   this._records = [];
 19 }
 20 
 21 /**
 22  * Define a name property.
 23  */
 24 ElementList.prototype.name = 'elementlist';
 25 
 26 /**
 27  * Returns the entire 'records' array.
 28  *
 29  * @returns {Array} An array of elements.
 30  */
 31 ElementList.prototype.add = function(obj) {
 32 
 33   'use strict';
 34 
 35   this._records.push(obj);
 36 
 37   return this._records;
 38 };
 39 
 40 /**
 41  * Returns the entire 'records' array.
 42  *
 43  * @returns {Array} An array of elements.
 44  */
 45 ElementList.prototype.all = function() {
 46   'use strict';
 47   return this._records;
 48 };
 49 
 50 /**
 51  * Returns the total number of elements.
 52  *
 53  * @returns {number} Total number of elements.
 54  */
 55 ElementList.prototype.count = function() {
 56   'use strict';
 57   return this._records.length;
 58 };
 59 
 60 /**
 61  * Returns an array of elements created from the same constructor.
 62  *
 63  * @param {string} name The constructor name.
 64  * @returns {Array} An array of elements.
 65  */
 66 ElementList.prototype.getAllByClass = function(name) {
 67 
 68   'use strict';
 69 
 70   var i, max, arr = [];
 71 
 72   for (i = 0, max = this._records.length; i < max; i++) {
 73     if (this._records[i].constructor.name === name) {
 74       arr[arr.length] = this._records[i];
 75     }
 76   }
 77   return arr;
 78 };
 79 
 80 /**
 81  * Updates the properties of elements created from the same constructor.
 82  *
 83  * @param {string} name The constructor name.
 84  * @param {Object} props A map of properties to update.
 85  * @returns {Array} An array of elements.
 86  * @example
 87  * exports.elementList.updatePropsByClass('Point', {
 88  *    color: [0, 0, 0],
 89  *    scale: 2
 90  * }); // all point will turn black and double in size
 91  */
 92 ElementList.prototype.updatePropsByClass = function(name, props) {
 93 
 94   'use strict';
 95 
 96   var i, max, p, arr = this.getAllByClass(name);
 97 
 98   for (i = 0, max = arr.length; i < max; i++) {
 99     for (p in props) {
100       if (props.hasOwnProperty(p)) {
101         arr[i][p] = props[p];
102       }
103     }
104   }
105   return arr;
106 };
107 
108 /**
109  * Finds an element by its 'id' and returns it.
110  *
111  * @param {string|number} id The element's id.
112  * @returns {Object} The element.
113  */
114 ElementList.prototype.getElement = function (id) {
115 
116   'use strict';
117 
118   var i, max, records = this._records;
119 
120   for (i = 0, max = records.length; i < max; i += 1) {
121     if (records[i].id === id) {
122       return records[i];
123     }
124   }
125 };
126 
127 /**
128  * Finds an element by its 'id' and removes it from its
129  * world as well as the records array.
130  *
131  * @param {string|number} id The element's id.
132  */
133 ElementList.prototype.destroyElement = function (id) {
134 
135   'use strict';
136 
137   var i, max, records = this._records;
138 
139   for (i = 0, max = records.length; i < max; i += 1) {
140     if (records[i].id === id) {
141       records[i].world.el.removeChild(records[i].el);
142       records.splice(i, 1);
143       break;
144     }
145   }
146 };
147 
148 /**
149  * Removes all elements from their world and resets
150  * the 'records' array.
151  *
152  * @param {string|number} id The element's id.
153  */
154 ElementList.prototype.destroyAll = function () {
155 
156   'use strict';
157 
158   var i, records = this._records;
159 
160   for (i = records.length - 1; i >= 0; i -= 1) {
161     if (records[i].world) {
162       records[i].world.el.removeChild(records[i].el);
163     }
164   }
165   this._records = [];
166 };
167 
168 /**
169  * Removes all elements from their world and resets
170  * the 'records' array.
171  *
172  * @param {string|number} id The element's id.
173  */
174 ElementList.prototype.destroyByWorld = function (world) {
175 
176   'use strict';
177 
178   var i, records = this._records;
179 
180   for (i = records.length - 1; i >= 0; i -= 1) {
181     if (records[i].world && records[i].world.id === world) {
182       records[i].world.el.removeChild(records[i].el);
183       records.splice(i, 1);
184     }
185   }
186 };
187 exports.ElementList = ElementList;
